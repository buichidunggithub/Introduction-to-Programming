    int a = 1, b = 1;
    for (int i = 1; i < n; ++i) {
        b = a + b;
        a = b - a;
    }
    return a;
}

int ptbac2(double a, double b, double c, double &x1, double &x2, int &sn)
{
	double delta;
	delta=b*b-4*a*c;
	if(delta<0) sn=0;
	else
	{
		if (delta==0)
			{
				sn=1;
				x1=x2=-b/(2*a);
			}
		else 
			{
				sn=2;
				x1=(-b+sqrt(delta))/(2*a);
				x2=(-b-sqrt(delta))/(2*a);
			}
	}	
	return 0;
}
